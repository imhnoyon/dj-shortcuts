from django.urls import path

    TokenObtainPairView,
    TokenRefreshView,
)

    # JWT Auth Endpoints

    # Core API Endpoints
    path('health/', HealthCheckView.as_view(), name='health_check'),
    path('users/me/', UserProfileView.as_view(), name='user_profile'),
